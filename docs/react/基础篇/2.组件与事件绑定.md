## 组件与 props

React 中，组件本质上就是函数，这个函数会返回一段 *JSX*，来表示这个组件的 *UI* 是什么样的。`props` 就是这个函数的输入，==由父组件传入，子组件只能读取，不应该直接修改==，这也是 React 中最核心的一条规则是

:::code-tabs
@tab App.jsx

```jsx
import ProfileCard from './ProfileCard';

export default function App() {
  return (
    <section>
      <ProfileCard
        name="ZhaoJisen"
        role="Frontend Engineer"
      >
        React、TypeScript、工程化
      </ProfileCard>
    </section>
  );
}
```

### 状态提升

当两个子组件需要共享一份数据时，不要试图让它们互相调用，而是把状态提到它们最近的共同父组件

:::code-tabs
@tab App.jsx

```jsx
import { useState } from 'react';
import MoneyInput from './MoneyInput';

const RATE = 7.2;

export default function App() {
  const [rmb, setRmb] = useState('');
  const [usd, setUsd] = useState('');

  function updateFromRmb(nextRmb) {
    setRmb(nextRmb);
    setUsd(nextRmb === '' ? '' : (Number(nextRmb) / RATE).toFixed(2));
  }

  function updateFromUsd(nextUsd) {
    setUsd(nextUsd);
    setRmb(nextUsd === '' ? '' : (Number(nextUsd) * RATE).toFixed(2));
  }

  return (
    <main>
      <MoneyInput label="人民币" value={rmb} onChange={updateFromRmb} />
      <MoneyInput label="美元" value={usd} onChange={updateFromUsd} />
    </main>
  );
}
```

@tab MoneyInput.jsx

```jsx
export default function MoneyInput({ label, value, onChange }) {
  function handleChange(event) {
    const nextValue = event.target.value;

    // 只允许空串或数字输入
    if (nextValue === '' || Number.isFinite(Number(nextValue))) {
      onChange(nextValue);
    }
  }

  return (
    <label style={{ display: 'block', marginBottom: 12 }}>
      <span style={{ display: 'inline-block', width: 72 }}>{label}</span>
      <input value={value} onChange={handleChange} />
    </label>
  );
}
```
